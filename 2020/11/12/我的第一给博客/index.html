– Table structure for book

DROP TABLE IF EXISTS book;
CREATE TABLE book (
bid int(11) NOT NULL,
bname varchar(50) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL,
author char(8) CHARACTER SET utf8 COLLATE utf8_general_ci DEFAULT NULL,
price float DEFAULT NULL,
publisher varchar(50) CHARACTER SET utf8 COLLATE utf8_general_ci DEFAULT NULL,
discount float DEFAULT NULL,
cid int(11) DEFAULT NULL,
PRIMARY KEY (bid) USING BTREE,
INDEX fk_book_category(cid) USING BTREE,
CONSTRAINT fk_book_category FOREIGN KEY (cid) REFERENCES category (cid) ON DELETE RESTRICT ON UPDATE RESTRICT
) ENGINE = InnoDB CHARACTER SET = utf8 COLLATE = utf8_general_ci ROW_FORMAT = Compact;

–(3)查询QQ邮箱

SELECT uname AS 会员昵称,email AS “E-mail”

–（4）订购日期

FROM b_order

–（6）名称价格

SELECT bname as 图书名称 , price as 价格

–项目2

–（1）图书最高
SELECT MAX( price) as 最高价格 ,MIN(price) as 最低价格,AVG(price) as 平均价格
FROM book

–（2）类别数量

SELECT bname as 图书名称,cid as 图书类别
FROM book

–（3）类别最高

SELECT cid, MAX( price) as 最高价格 ,MIN(price) as 最低价格,AVG(price) as 平均价格
FROM book
GROUP BY cid
